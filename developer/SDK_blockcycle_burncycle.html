2BCycle Architecture

BlockCycle (Backend) + BurnCycle (Frontend) - Pendekatan Terpisah untuk Efisiensi dan Skalabilitas

BlockCycle (Backend)

Fokus pada logika bisnis, validasi data, penyimpanan, dan eksekusi operasi blockchain. Bertanggung jawab atas keamanan, konsistensi, dan manajemen data.

BurnCycle (Frontend)

Fokus pada pengalaman pengguna, presentasi data, interaksi, dan visualisasi. Bertanggung jawab atas responsivitas, aksesibilitas, dan kemudahan penggunaan.

Prinsip Integrasi

Komunikasi melalui API yang terdefinisi dengan baik. Frontend hanya berinteraksi dengan backend melalui endpoint yang telah ditentukan, menjaga separasi yang jelas.

BlockCycle API (Backend)
Solidity + Node.js 
// Endpoint: POST /api/execute-command
router.post('/execute-command', async (req, res) => {
    // Validasi input dari frontend
    const { command, params, walletAddress } = req.body;
    
    try {
        // Validasi struktur perintah
        if (!validateCommandSchema(command, params)) {
            return res.status(400).json({ error: "Invalid command structure" });
        }
        
        // Verifikasi kepemilikan wallet
        if (!verifyWalletOwnership(walletAddress)) {
            return res.status(403).json({ error: "Wallet verification failed" });
        }
        
        // Proses perintah di blockchain
        const txHash = await blockchainService.executeCommand(
            command, 
            params, 
            walletAddress
        );
        
        // Simpan log transaksi
        await db.saveTransactionLog({
            command,
            params,
            walletAddress,
            txHash,
            status: "pending"
        });
        
        // Kirim respons ke frontend
        res.json({ 
            success: true, 
            txHash,
            message: "Command executed successfully"
        });
        
    } catch (error) {
        console.error("API Error:", error);
        res.status(500).json({ 
            error: "Internal server error",
            details: error.message 
        });
    }
});
BurnCycle UI (Frontend)
React + Ethers.js 
// Komponen Terminal UI
function CommandTerminal() {
    const [input, setInput] = useState('');
    const [output, setOutput] = useState([]);
    const { wallet } = useWallet();
    
    const executeCommand = async (command) => {
        // Parse perintah pengguna
        const parsed = parseCommand(command);
        
        try {
            // Tampilkan perintah di output
            addOutput(`> ${command}`, 'command');
            
            // Kirim ke backend BlockCycle
            const response = await api.post('/execute-command', {
                command: parsed.action,
                params: parsed.params,
                walletAddress: wallet.address
            });
            
            if (response.success) {
                addOutput(`[✔] ${response.message}`, 'success');
                addOutput(`Transaction: ${response.txHash}`, 'info');
            } else {
                addOutput(`[✖] Error: ${response.error}`, 'error');
            }
            
        } catch (error) {
            addOutput(`[✖] API Error: ${error.message}`, 'error');
        }
    };
    
    const handleSubmit = (e) => {
        e.preventDefault();
        executeCommand(input);
        setInput('');
    };
    
    return (
        <div className="terminal">
            <div className="output">
                {output.map((item, i) => (
                    <pre key={i} className={item.type}>{item.text}</pre>
                ))}
            </div>
            <form onSubmit={handleSubmit}>
                <input 
                    type="text"
                    value={input}
                    onChange={(e) => setInput(e.target.value)}
                    placeholder="Enter command..."
                />
                <button type="submit">Execute</button>
            </form>
        </div>
    );
}
